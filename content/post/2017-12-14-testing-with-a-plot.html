rladies_meetup_groups <- read_csv("../../data/rladies_meetup_groups.csv")

rladies_meetup_groups <- rladies_meetup_groups %>%
  separate(timezone, c("region"), sep="/", extra="drop", fill="left")

ggplot(rladies_meetup_groups ,
       aes(x = reorder(localized_location, members), y = members, fill = region)) +
  theme_few() +
  geom_col(position="stack") +
  coord_flip() +
  labs(title="R-Ladies Meetup Membership - November 2017", x="Location", y="R-Ladies")
